JavaScript // SEOBAIKE API — 服務查詢範例

const SEOBAIKE_API = 'https://aiforseo.vip/api/v1';

async function queryServices(category) {
  const res = await fetch(
    `${SEOBAIKE_API}/services?cat=${category}`,
    {
      headers: {
        'Authorization': 'Bearer YOUR_API_KEY',
        'Content-Type': 'application/json'
      }
    }
  );

  const data = await res.json();
  return data;
}

// 回傳格式
// {
// "services": [
// { "id": "...", "name": "...", "status": "active" }
// ],
// "total": 42,
// "page": 1
